using System.Web.OData.Builder;

    public static class WebApiConfig
    {
        public static void Register(HttpConfiguration config)
        {
            var allowedCors = new EnableCorsAttribute("*", "*", "*");
            config.EnableCors(allowedCors);

            config.MapHttpAttributeRoutes();

            //// Web API routes

            //config.Routes.MapHttpRoute(

các câu truy vấn $expand,$select,$value trong web api ODATA

$expand=?? (Mở rộng các thực thể liên quan trong object)

$select=?? (Chọn một vài trường để trả về)

$value=?? Chỉ trả về giá trị của Object ví dụ : {Id:1, Name:"MR Tai"} sử dụng $value trả về 1,Mr Tai
